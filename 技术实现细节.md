### 1. 用户类型分析算法

```javascript
// 多维度权重分析算法
function analyzeUserType(userData) {
    const preferences = userData.preferences || {};
    const marketPrefs = userData.marketPreferences || {};
    
    // 权重配置
    const weights = {
        travelStyle: 0.30,    // 旅行风格权重30%
        budget: 0.20,         // 预算偏好权重20%
        riskTolerance: 0.25,  // 风险偏好权重25%
        interests: 0.15,      // 兴趣标签权重15%
        other: 0.10          // 其他因素权重10%
    };
    
    let userType = '理财型游客'; // 默认类型
    let confidence = 0;
    
    // 旅行风格分析
    if (preferences.travelStyle === '商务') {
        userType = '轻旅行打工人';
        confidence += weights.travelStyle * 100;
    } else if (preferences.travelStyle === '家庭') {
        userType = '家庭度假型';
        confidence += weights.travelStyle * 100;
    } else if (preferences.travelStyle === '康养') {
        userType = '康养型用户';
        confidence += weights.travelStyle * 100;
    } else if (preferences.travelStyle === '探险') {
        userType = '探索型年轻人';
        confidence += weights.travelStyle * 100;
    }
    
    // 预算偏好分析
    if (preferences.budget === '高档') {
        if (userType === '理财型游客') confidence += weights.budget * 100;
    } else if (preferences.budget === '经济') {
        if (userType === '轻旅行打工人') confidence += weights.budget * 100;
    }
    
    // 风险偏好分析
    if (marketPrefs.riskTolerance === 'low') {
        if (userType === '理财型游客') confidence += weights.riskTolerance * 100;
    } else if (marketPrefs.riskTolerance === 'high') {
        if (userType === '探索型年轻人') confidence += weights.riskTolerance * 100;
    }
    
    // 兴趣标签分析
    if (preferences.interests && preferences.interests.includes('文化')) {
        if (userType === '理财型游客') confidence += weights.interests * 100;
    }
    if (preferences.interests && preferences.interests.includes('美食')) {
        if (userType === '家庭度假型') confidence += weights.interests * 100;
    }
    
    return { userType, confidence: Math.min(confidence, 100) };
}
```

### 2. 数据聚合算法

```javascript
// 用户分群聚合算法
function aggregateUserSegments(userDataList) {
    const segments = {
        '理财型游客': { n: 0, percentage: 0 },
        '轻旅行打工人': { n: 0, percentage: 0 },
        '家庭度假型': { n: 0, percentage: 0 },
        '康养型用户': { n: 0, percentage: 0 },
        '探索型年轻人': { n: 0, percentage: 0 }
    };
    
    // 统计各类型用户数量
    userDataList.forEach(user => {
        const analysis = analyzeUserType(user);
        if (segments[analysis.userType]) {
            segments[analysis.userType].n++;
        }
    });
    
    const total = userDataList.length;
    
    // 计算百分比
    Object.keys(segments).forEach(type => {
        segments[type].percentage = total > 0 ? 
            ((segments[type].n / total) * 100).toFixed(1) : 0;
    });
    
    return {
        total,
        segments: Object.keys(segments).map(label => ({
            label,
            n: segments[label].n,
            percentage: parseFloat(segments[label].percentage)
        }))
    };
}
```

### 3. 实时数据更新机制

```javascript
// 实时数据更新
class UserPortraitManager {
    constructor() {
        this.updateInterval = 30000; // 30秒更新一次
        this.isUpdating = false;
    }
    
    startRealTimeUpdate() {
        setInterval(() => {
            if (!this.isUpdating) {
                this.updateUserPortraits();
            }
        }, this.updateInterval);
    }
    
    async updateUserPortraits() {
        this.isUpdating = true;
        try {
            // 获取最新用户数据
            const userData = await this.fetchUserData();
            
            // 重新分析用户画像
            const analysis = analyzeUserType(userData);
            
            // 更新本地存储
            this.updateLocalStorage(analysis);
            
            // 触发UI更新
            this.triggerUIUpdate(analysis);
            
        } catch (error) {
            console.error('用户画像更新失败:', error);
        } finally {
            this.isUpdating = false;
        }
    }
    
    triggerUIUpdate(analysis) {
        // 触发自定义事件，通知UI更新
        const event = new CustomEvent('portraitUpdated', {
            detail: analysis
        });
        document.dispatchEvent(event);
    }
}
```

```javascript
// WebSocket 实时数据推送
class RealTimeDataManager {
    constructor() {
        this.ws = null;
        this.reconnectAttempts = 0;
        this.maxReconnectAttempts = 5;
    }
    
    connect() {
        this.ws = new WebSocket('ws://localhost:3000/realtime');
        
        this.ws.onopen = () => {
            console.log('实时数据连接已建立');
            this.reconnectAttempts = 0;
        };
        
        this.ws.onmessage = (event) => {
            const data = JSON.parse(event.data);
            this.handleRealtimeData(data);
        };
        
        this.ws.onclose = () => {
            console.log('实时数据连接已断开');
            this.reconnect();
        };
        
        this.ws.onerror = (error) => {
            console.error('实时数据连接错误:', error);
        };
    }
    
    handleRealtimeData(data) {
        switch (data.type) {
            case 'userPortraitUpdate':
                this.updateUserPortrait(data.payload);
                break;
            case 'segmentUpdate':
                this.updateSegmentData(data.payload);
                break;
            case 'analyticsUpdate':
                this.updateAnalytics(data.payload);
                break;
        }
    }
    
    reconnect() {
        if (this.reconnectAttempts < this.maxReconnectAttempts) {
            this.reconnectAttempts++;
            setTimeout(() => {
                console.log(`尝试重连 (${this.reconnectAttempts}/${this.maxReconnectAttempts})`);
                this.connect();
            }, 2000 * this.reconnectAttempts);
        }
    }
}
```

```javascript
class VirtualScrollManager {
    constructor(container, itemHeight, data) {
        this.container = container;
        this.itemHeight = itemHeight;
        this.data = data;
        this.visibleCount = Math.ceil(container.clientHeight / itemHeight);
        this.scrollTop = 0;
        
        this.init();
    }
    
    init() {
        this.container.addEventListener('scroll', this.handleScroll.bind(this));
        this.render();
    }
    
    handleScroll() {
        this.scrollTop = this.container.scrollTop;
        this.render();
    }
    
    render() {
        const startIndex = Math.floor(this.scrollTop / this.itemHeight);
        const endIndex = Math.min(startIndex + this.visibleCount, this.data.length);
        
        const visibleData = this.data.slice(startIndex, endIndex);
        
        // 只渲染可见区域的数据
        this.renderVisibleItems(visibleData, startIndex);
    }
}
```

### 2. 数据缓存策略

```javascript
// LRU缓存实现
class LRUCache {
    constructor(capacity) {
        this.capacity = capacity;
        this.cache = new Map();
    }
    
    get(key) {
        if (this.cache.has(key)) {
            const value = this.cache.get(key);
            // 移动到末尾（最近使用）
            this.cache.delete(key);
            this.cache.set(key, value);
            return value;
        }
        return null;
    }
    
    set(key, value) {
        if (this.cache.has(key)) {
            this.cache.delete(key);
        } else if (this.cache.size >= this.capacity) {
            // 删除最久未使用的项
            const firstKey = this.cache.keys().next().value;
            this.cache.delete(firstKey);
        }
        this.cache.set(key, value);
    }
}
```

### 1. 数据加密

```javascript
// 前端数据加密
class DataEncryption {
    constructor() {
        this.key = this.generateKey();
    }
    
    generateKey() {
        // 生成随机密钥
        return crypto.getRandomValues(new Uint8Array(32));
    }
    
    encrypt(data) {
        const encoder = new TextEncoder();
        const dataBuffer = encoder.encode(JSON.stringify(data));
        
        return crypto.subtle.encrypt(
            { name: 'AES-GCM', iv: crypto.getRandomValues(new Uint8Array(12)) },
            this.key,
            dataBuffer
        );
    }
    
    decrypt(encryptedData) {
        return crypto.subtle.decrypt(
            { name: 'AES-GCM', iv: encryptedData.iv },
            this.key,
            encryptedData.data
        );
    }
}
```

```javascript
// 数据脱敏处理
class DataMasking {
    static maskUserData(userData) {
        const masked = { ...userData };
        
        // 脱敏处理
        if (masked.phone) {
            masked.phone = masked.phone.replace(/(\d{3})\d{4}(\d{4})/, '$1****$2');
        }
        
        if (masked.email) {
            const [username, domain] = masked.email.split('@');
            masked.email = username.charAt(0) + '***@' + domain;
        }
        
        if (masked.idCard) {
            masked.idCard = masked.idCard.replace(/(\d{6})\d{8}(\d{4})/, '$1********$2');
        }
        
        return masked;
    }
    
    static anonymizePortraitData(portraitData) {

            confidence: portraitData.confidence,
```

```javascript
// 性能监控
class PerformanceMonitor {
    constructor() {
        this.metrics = {
            responseTime: [],
            errorRate: 0,
            throughput: 0
        };
    }
    
    trackResponseTime(startTime) {
        const responseTime = Date.now() - startTime;
        this.metrics.responseTime.push(responseTime);
        
        // 保持最近1000个记录
        if (this.metrics.responseTime.length > 1000) {
            this.metrics.responseTime.shift();
        }
    }
    
    getAverageResponseTime() {
        if (this.metrics.responseTime.length === 0) return 0;
        return this.metrics.responseTime.reduce((a, b) => a + b, 0) / this.metrics.responseTime.length;
    }
    
    reportMetrics() {
        return {
            averageResponseTime: this.getAverageResponseTime(),
            errorRate: this.metrics.errorRate,
            throughput: this.metrics.throughput,
            timestamp: new Date().toISOString()
        };
    }
}
```
